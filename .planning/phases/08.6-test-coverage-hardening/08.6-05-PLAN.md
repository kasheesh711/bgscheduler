import { describe, it, expect, beforeAll, afterAll, beforeEach } from "vitest";
import { eq } from "drizzle-orm";
import { startTestDb, stopTestDb, truncateAll } from "@/tests/integration/db-helper";
import { runPastSessionsDiffHook } from "@/lib/sync/past-sessions-diff-hook";
import * as schema from "@/lib/db/schema";
import type { NormalizedSessionBlock } from "@/lib/normalization/sessions";

let handle: Awaited<ReturnType<typeof startTestDb>>;

beforeAll(async () => { handle = await startTestDb(); }, 60_000);
afterAll(async () => { if (handle) await stopTestDb(handle); });
beforeEach(async () => { await truncateAll(handle.db); });

/**
 * Helper: seed a prior active snapshot with N past sessions + M future sessions
 * tied to a single identity group with the given canonicalKey.
 */
async function seedPriorSnapshot(opts: {
  groupCanonicalKey: string;
  pastWiseSessionIds: string[];
  futureWiseSessionIds: string[];
}): Promise<{ snapshotId: string; groupId: string }> {
  const [snap] = await handle.db.insert(schema.snapshots).values({ active: true })
    .returning({ id: schema.snapshots.id });

  const [group] = await handle.db.insert(schema.tutorIdentityGroups).values({
    snapshotId: snap.id,
    canonicalKey: opts.groupCanonicalKey,
    displayName: "Test Tutor",
    supportedModality: "onsite",
  }).returning({ id: schema.tutorIdentityGroups.id });

  const PAST = new Date(Date.now() - 24 * 3600_000);
  const FUTURE = new Date(Date.now() + 24 * 3600_000);

  const rows: typeof schema.futureSessionBlocks.$inferInsert[] = [
    ...opts.pastWiseSessionIds.map((wid) => ({
      snapshotId: snap.id,
      groupId: group.id,
      wiseTeacherId: "t1",
      wiseSessionId: wid,
      startTime: PAST,
      endTime: new Date(PAST.getTime() + 3600_000),
      weekday: PAST.getDay(),
      startMinute: 600,
      endMinute: 660,
      wiseStatus: "CONFIRMED",
      isBlocking: true,
    })),
    ...opts.futureWiseSessionIds.map((wid) => ({
      snapshotId: snap.id,
      groupId: group.id,
      wiseTeacherId: "t1",
      wiseSessionId: wid,
      startTime: FUTURE,
      endTime: new Date(FUTURE.getTime() + 3600_000),
      weekday: FUTURE.getDay(),
      startMinute: 600,
      endMinute: 660,
      wiseStatus: "CONFIRMED",
      isBlocking: true,
    })),
  ];
  if (rows.length) await handle.db.insert(schema.futureSessionBlocks).values(rows);

  return { snapshotId: snap.id, groupId: group.id };
}

describe("runPastSessionsDiffHook — TCOV-04 integration (real Postgres)", () => {
  it("captures dropped past sessions into past_session_blocks", async () => {
    await seedPriorSnapshot({
      groupCanonicalKey: "tutor-a",
      pastWiseSessionIds: ["S-past-1", "S-past-2"],
      futureWiseSessionIds: ["S-future-1"],
    });

    // New snapshot row (not yet active)
    const [newSnap] = await handle.db.insert(schema.snapshots).values({ active: false })
      .returning({ id: schema.snapshots.id });

    // New Wise response only contains S-future-1 — past ones dropped
    const newSessions: NormalizedSessionBlock[] = [{
      wiseSessionId: "S-future-1",
      wiseTeacherId: "t1",
      startTime: new Date(Date.now() + 24 * 3600_000),
      endTime: new Date(Date.now() + 25 * 3600_000),
      weekday: 0,
      startMinute: 600,
      endMinute: 660,
      wiseStatus: "CONFIRMED",
      isBlocking: true,
    }];

    const result = await runPastSessionsDiffHook(handle.db as never, newSessions, newSnap.id);

    expect(result.capturedCount).toBe(2);

    // Real schema types — these assertions now have correct types,
    // resolving the TS errors at the old test's lines 185-186.
    const captured = await handle.db.select().from(schema.pastSessionBlocks);
    expect(captured.map((r) => r.wiseSessionId).sort()).toEqual(["S-past-1", "S-past-2"]);
    expect(captured.every((r) => r.groupCanonicalKey === "tutor-a")).toBe(true);
    expect(captured.every((r) => r.capturedInSnapshotId === newSnap.id)).toBe(true);
  });

  it("idempotent: second run doesn't double-insert (UNIQUE constraint)", async () => {
    await seedPriorSnapshot({
      groupCanonicalKey: "tutor-a",
      pastWiseSessionIds: ["S1"],
      futureWiseSessionIds: [],
    });
    const [newSnap] = await handle.db.insert(schema.snapshots).values({ active: false })
      .returning({ id: schema.snapshots.id });

    const r1 = await runPastSessionsDiffHook(handle.db as never, [], newSnap.id);
    expect(r1.capturedCount).toBe(1);

    const r2 = await runPastSessionsDiffHook(handle.db as never, [], newSnap.id);
    expect(r2.capturedCount).toBe(0); // ON CONFLICT DO NOTHING

    const allCaptured = await handle.db.select().from(schema.pastSessionBlocks)
      .where(eq(schema.pastSessionBlocks.wiseSessionId, "S1"));
    expect(allCaptured).toHaveLength(1);
  });

  it("future sessions are never captured even when dropped", async () => {
    await seedPriorSnapshot({
      groupCanonicalKey: "tutor-a",
      pastWiseSessionIds: [],
      futureWiseSessionIds: ["S-still-future-1"], // future session in prior snapshot
    });
    const [newSnap] = await handle.db.insert(schema.snapshots).values({ active: false })
      .returning({ id: schema.snapshots.id });

    // New response drops the future session entirely
    const result = await runPastSessionsDiffHook(handle.db as never, [], newSnap.id);

    // Future-dated session must NOT be captured (only past-dated drops are)
    expect(result.capturedCount).toBe(0);
    const captured = await handle.db.select().from(schema.pastSessionBlocks);
    expect(captured).toHaveLength(0);
  });

  it("orphan groupId emits completeness data_issue without throwing", async () => {
    // Seed a prior snapshot, then DELETE the identity group (simulates a
    // group that existed in the prior snapshot but is gone from priorGroups
    // when the diff hook runs — orphan reference).
    const { snapshotId, groupId } = await seedPriorSnapshot({
      groupCanonicalKey: "tutor-orphan",
      pastWiseSessionIds: ["S-orphan-1"],
      futureWiseSessionIds: [],
    });

    // Delete the group — leaves the futureSessionBlock row dangling
    await handle.db.delete(schema.tutorIdentityGroups)
      .where(eq(schema.tutorIdentityGroups.id, groupId));

    const [newSnap] = await handle.db.insert(schema.snapshots).values({ active: false })
      .returning({ id: schema.snapshots.id });

    // Hook must NOT throw; should emit a completeness data_issue
    const result = await runPastSessionsDiffHook(handle.db as never, [], newSnap.id);
    expect(result).toBeDefined(); // did not throw

    // A completeness data_issue row should exist
    const issues = await handle.db.select().from(schema.dataIssues)
      .where(eq(schema.dataIssues.type, "completeness"));
    expect(issues.length).toBeGreaterThanOrEqual(1);
  });
});
