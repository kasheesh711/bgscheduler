          qualifications: [],

    expect(body).toMatchObject({
      snapshotMeta: expect.objectContaining({ snapshotId: "snap-1" }),
      tutors: expect.any(Array),
      conflicts: expect.any(Array),
      sharedFreeSlots: expect.any(Array),
    });
  });
