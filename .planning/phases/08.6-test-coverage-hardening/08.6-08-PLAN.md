    // Inspect the values payloads — at least one must contain a row with type='conflict_model'

    expect(conflictCalls.length).toBeGreaterThan(0);

  it("populates metadata field with isOnlineVariant, sessionType, and groupCanonicalKey", async () => {

    // Find the conflict_model row and inspect its metadata
    let conflictRow: any = null;
    for (const [rows] of valuesSpy.mock.calls) {
      if (Array.isArray(rows)) {
        const found = rows.find((r: any) => r.type === "conflict_model");
        if (found) {
          conflictRow = found;
          break;
        }

    expect(conflictRow).not.toBeNull();
    expect(conflictRow.metadata).toBeDefined();
    expect(conflictRow.metadata.isOnlineVariant).toBeDefined();
    expect(conflictRow.metadata.sessionType).toBeDefined();
    expect(conflictRow.metadata.groupCanonicalKey).toBeDefined();

  it("does NOT emit conflict_model when modality is consistent", async () => {
    const { db, valuesSpy } = makeDbStub();

    await runFullSync(db, makeConsistentClient(), "test-institute-id");

      Array.isArray(rows) && rows.some((r: any) => r.type === "conflict_model")
    );
    expect(conflictCalls.length).toBe(0);
  });
});
