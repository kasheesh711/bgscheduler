interface FakeTablesState {

function makeFakeDbWithTables(state: FakeTablesState): Database {
  function chain(target: unknown) {
    const limitFn = vi.fn().mockImplementation(async () => {
      // Active-snapshot lookup (limit(1)) returns [{ id }] when set
      if (state.activeSnapshotId !== null) return [{ id: state.activeSnapshotId }];
      return [];
    });
    const whereChain = {
      limit: limitFn,
      then: (onFulfilled: (rows: unknown[]) => unknown) => {
        const rows = state.rowsByTable.get(target) ?? [];
        return Promise.resolve(rows).then(onFulfilled);
      },
    };
    const whereFn = vi.fn().mockReturnValue(whereChain);
    const fromFn = vi.fn().mockImplementation((target: unknown) => {
      // Capture the target so the where->then can dispatch by table
      return { where: whereFn, _target: target };
    });
    // Bind the captured target through to the where chain.
    // Simpler: dispatch directly via the from impl below.
    return { from: vi.fn().mockImplementation((tgt: unknown) => ({
      where: vi.fn().mockReturnValue({
        limit: limitFn,
        then: (onFulfilled: (rows: unknown[]) => unknown) => {
          const rows = state.rowsByTable.get(tgt) ?? [];
          return Promise.resolve(rows).then(onFulfilled);
        },
      }),
    })) };
  }

  const selectFn = vi.fn().mockImplementation(() => {
    state.selectCallCount += 1;
    return chain(null);
  });

  return { select: selectFn } as unknown as Database;
}
