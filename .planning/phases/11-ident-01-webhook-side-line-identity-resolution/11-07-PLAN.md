      result: { followerCount: number; upsertedContacts: number; suggestionsCreated: number };

    void Promise.all([loadRuns(), loadSummary()]);
  } catch (error) {
    setMessage(error instanceof Error ? error.message : "Re-anchor failed");
  } finally {
    setBusy(null);
  }
}, [loadRuns, loadSummary]);
